      noCount++;

 

      noBtn.style.left = Math.random() * 85 + "vw";

      noBtn.style.top = Math.random() * 85 + "vh";

      noBtn.style.transform = `scale(${Math.max(0.4, 1 - noCount * 0.1)})`;

    });
